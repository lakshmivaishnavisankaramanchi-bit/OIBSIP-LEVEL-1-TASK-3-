// Function to convert temperature function convertTemp() { // Get input value from user let tempInput = document.getElementById("tempInput"); let tempValue = tempInput.value; // Get selected conversion unit let unit = document.getElementById("unit").value; // Result display element let result = document.getElementById("result"); // Validation: check if input is empty if (tempValue === "") { result.innerHTML = "❌ Please enter a temperature value"; return; } // Convert input value to number let temperature = Number(tempValue); // Validation: check if input is a valid number if (isNaN(temperature)) { result.innerHTML = "❌ Invalid input. Please enter a number."; return; } // Perform conversion based on selected unit if (unit === "celsius") { // Celsius to Fahrenheit formula let fahrenheit = (temperature * 9 / 5) + 32; // Display result result.innerHTML = temperature + " °C is equal to " + fahrenheit.toFixed(2) + " °F"; } else if (unit === "fahrenheit") { // Fahrenheit to Celsius formula let celsius = (temperature - 32) * 5 / 9; // Display result result.innerHTML = temperature + " °F is equal to " + celsius.toFixed(2) + " °C"; } else { // Safety fallback (rare case) result.innerHTML = "⚠️ Please select a conversion type"; } }
